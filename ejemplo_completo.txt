Clase EjemploCompleto
    Funcion ordenarBurbuja(arreglo, n)
        Para i desde 1 hasta n hacer
            Para j desde 1 hasta n-i hacer
                Si (arreglo[j] > arreglo[j+1]) Entonces
                    temp <- arreglo[j]
                    arreglo[j] <- arreglo[j+1]
                    arreglo[j+1] <- temp
                fSi
            fPara
        fPara
    fFuncion
    
    Funcion buscarElemento(arreglo, elemento, tamaño)
        encontrado <- 0
        posicion <- -1
        Para i desde 1 hasta tamaño hacer
            Si (arreglo[i] = elemento) Entonces
                encontrado <- 1
                posicion <- i
            fSi
        fPara
        retornar posicion
    fFuncion
fClase
